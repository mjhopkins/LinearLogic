$\perp$

\begin{prooftree}
\def\fCenter{\mbox{\ $\vdash$\ }}
\Axiom   $\Gamma          \fCenter A, \Delta$

\begin{prooftree}
\def\fCenter{\mbox{\ $\vdash$\ }}
\Axiom    $\Gamma, A \fCenter \Delta          $
\UnaryInf $\Gamma    \fCenter A^\perp , \Delta$
\end{prooftree}

$\with$
